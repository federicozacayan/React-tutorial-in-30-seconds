How to update an Array State

We have only one method to update specific state. So, if we work with arrays we have to use array functions avoiding update the current state. We just have to replace it. That is why we set a new array.

import React, { useState } from 'react';
export default function Body() {
    const [count, setCount] = useState(['a', 'b', 'c', 'd']);
    function addElement (){
        const newCount = count.concat('x')
        //const newCount = [...count, 'x'];
        setCount(newCount)
    }
    function deleteByValue (){
        const newCount = count.filer(n => n !== 'x');
        setCount(newCount)
    }
    function deleteByPosition (){
        const newCount = count.map(n => n);
        newCount.splice(1, 0)
        setCount(newCount)
    }
    function replacingByPosition (){
        const newCount = count.map(n => n);
        newCount.splice(1, 1, 'y')
        setCount(newCount)
    }
    return (
        <div>
            Count : {count.join(',')}
            <input type="button" onClick={addElement} value="Add element" />
            <input type="button" onClick={deleteByValue} value="Delete by value" />
            <input type="button" onClick={deleteByPosition} value="Delete by position" />
            <input type="button" onClick={replacingByPosition} value="Replacing by position" />
        </div>
    )
}

LocalStorage in React

We can persist data in the browser using localStorage. Unfortunately, it save only strings, so we have to use JSON.stringify() and JSON.parse()

import React, { useState, useEffect } from 'react';
export default function Body() {
    const [count, setCount] =
        useState(JSON.parse(localStorage.getItem('count')) || 0);
    function increment (){
        setCount(count);
    }
    useEffect(() => {
        localStorage.setItem('count', JSON.stringify(count));
    }, [count]);
    return (
        <div>
            Count : {count}
            <input type="button" onClick={increment} value="increase" />
        </div>
    )
}

How to avoid re-execution

Using useEffect() function with no dependent variable we can run a piece of code only the first time the component is rendered.

import React, { useState, useEffect } from "react";
import User from './User'
function App() {
    const [info, setInfo] = useState();
    useEffect(() => {
        fetch('http://jsonplaceholder.typicode.com/users')
        .then(res => res.json())
        .then((data) => {
            console.log(data);
            setInfo(data)
        })
        .catch(console.log)
    }, []);//<-- no dependent variable
    if(typeof info === 'undefined'){
        return (<div>Loading...</div>)
    } else {
        return (<div>
            {info.map((contact) => (
                <User key={contact.id} contact={contact}></User>
            ))}
            </div>)
    }
}
export default App;

How to refer to an Element

We can use the ref attribute to get a pointer to an element itself. 

import React from 'react';
export default function Body() {
    let specialElement1 = null;
    let specialElement2 = null;   
    function focus1 (){
        specialElement1.focus();
    }
    let focus2 = () =>{
        specialElement2.focus();
    }
    return (
        <div>
            <input type="button" onClick={focus1} value="Focus Input 1" />
            <input type="button" onClick={focus2} value="Focus Input 2"/>
            <input type="text" value="Input 1" ref={elem => (specialElement1 = elem)} />
            <input type="text" value="Input 2" ref={elem => (specialElement2 = elem)} />
        </div>
    )
}

Context

How to share data between components.

We need several things:

1) A Context Component where place the data and its update methods, witch is a kind of <CONTAINER-VARIABLE>.
2) A <PROVIDER> (Container Component) to retrieve data and configure update methods.
3) One or many <CONSUMERS> (Representational Components) to show that data and execute its update methods.

The Consumer have to be a sub component of a Provider to receive a context from it.

/* src/context/ThemeContext.js */
import {createContext} from 'react';
const ThemeContext = createContext(null);
export default ThemeContext;
/* src/component/App.js */
import React, {useState} from 'react';
import ThemeContext from '../context/ThemeContext';
import Consumer from './Consumer';
function App() {
  const [val, setVal] = useState(0);
  let obj = {val, setVal}
  return (
    <ThemeContext.Provider value={obj} >
      <Consumer />
    </ThemeContext.Provider>
  );
}
export default App;

The context receive a particular "props", named value where is place the object to share with the sub components.

/* src/component/comsumer.js */
import React, {useEffect} from 'react'
import ThemeContext from '../context/ThemeContext';
export default function Consumer() {
    return (
        <ThemeContext.Consumer>
            {(obj) => (
                <div>
                    <p>{obj.val}</p>
                    <input type="button" value="increment" onClick={()=>{obj.setVal(obj.val+1)}}/>
                </div>
            )}
        </ThemeContext.Consumer>
    )
}

Redux

It looks like a different topic, but it is the same concept of Context, it just has additional implementations.

Context has:
1) <CONTAINER-VARIABLE>
2) <PROVIDER>
3) <CONSUMER>
Redux has:
1) Many <CONTAINER-VARIABLE>, plus channels, plus default value.
2) <PROVIDER> replacing "value" props by "store" props instead.
3) <CONSUMER> consuming different <CONTAINER-VARIABLE>s.
4) <BROADCAST> with a channel and none, one or many params.

Let's start it

npm install redux react-redux

Actions <BROADCAST>

/* src/actions/index.js */
// one action one channel
export const f1 = (number) => {
    return {
        type: 'A', //channel A
        parameter: number
    }
}
// other action one channel
export const f2 = () => {
    return {
        type: 'B' //channel B
    }
}

Reducers <CONTAINER-VARIABLE>

/* src/reducers/counter.js */
// the behaviour depends on the channel and params.
const counterReducer =  (state = 2, action)=>{
    switch(action.type){
        case 'A':
            if(action.parameter){ // optional param
                return state + action.parameter; 
            }
            return state;
        case 'B':
            return state-1;
        default:
            return state;
    }
}
export default counterReducer;
/* src/reducers/isloged.js */
// the behaviour depends on the channel and param.
const loggedReducer =  (state = false, action) => {
    switch(action.type){
        case 'A':
            return true;
        case 'B':
            return false;
        default:
            return state;
    }
}
export default loggedReducer;

List of Reducers

The reducers are joined in a specific reducer container. 

/* src/reducers/index.js */
import counterReducer from "./counter";
import loggedReducer from "./isloged";
import {combineReducers} from 'redux'

const reducers = combineReducers({
    counter: counterReducer,
    islooged: loggedReducer
});

export default reducers;

ReactDOM.render(

Consumer <CONSUMER>

/* src/components/Container.js */
import React from 'react'
import Button from './Button' // Reusable component

import { useSelector, useDispatch } from 'react-redux'
import {f1, f2} from '../actions/index'

export default function Container() {
  const counter = useSelector(state=>state.counter);
  const islooged = useSelector(state=>state.islooged);
  const reduxManager = useDispatch();
  return (
    <div>
      <Button onClick={()=> reduxManager(f1(5))} text="add" />
      <Button onClick={()=> reduxManager(f2())} text="take" />
      {counter} - {(islooged)?'true':'false'}
    </div>
  )
}
